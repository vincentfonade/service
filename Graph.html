</head>

<body>

    <!-- Canvas pour le graphique radar -->

    <canvas id="radarChart" width="600" height="400"></canvas>


    <script>

        // Fonction pour récupérer les paramètres dynamiques de l'URL

        function getUrlParameter(name) {

            const urlParams = new URLSearchParams(window.location.search);

            const param = urlParams.get(name);

            return param ? param.split(',').map(Number) : null;

        }


        // Récupérer les valeurs dynamiques de l'URL

        // Exemple : ?ResultatEntreprise=2,3,1,4,5,2,1,3

        const dynamicValues = getUrlParameter('ResultatEntreprise') || [2, 3, 1, 4, 5, 2, 1, 3]; // Valeurs par défaut


        // Libellés des axes (adaptés à votre exemple)

        const labels = [

            "Élaborer vos stratégies RH",

            "Manager et accompagner",

            "Donner du sens",

            "Mieux vivre ensemble",

            "Grandir ensemble",

            "Proposer un package de rémunération attractif",

            "Répondre aux exigences légales",

            "Mieux vivre ensemble"

        ];


        // Configuration du graphique

        const data = {

            labels: labels,

            datasets: [{

                label: "Score",

                data: dynamicValues, // Utilisation des valeurs dynamiques

                fill: true,

                backgroundColor: "rgba(75, 192, 192, 0.2)",

                borderColor: "rgba(75, 192, 192, 1)",

                pointBackgroundColor: "rgba(75, 192, 192, 1)",

                pointBorderColor: "#fff",

                pointHoverBackgroundColor: "#fff",

                pointHoverBorderColor: "rgba(75, 192, 192, 1)"

            }]

        };


        const config = {

            type: 'radar',

            data: data,

            options: {

                responsive: true,

                scales: {

                    r: {

                        beginAtZero: true,

                        max: 5

                    }

                }

            }

        };


        // Rendu du graphique

        const radarChart = new Chart(

            document.getElementById('radarChart'),

            config

        );

    </script>
